// PredictionMarket.sol // Main contract — orchestrates markets, pools, payouts import "./AMMPricer.sol"; struct Market { uint id; string question; uint endTime; bool resolved; bool outcome; // true = YES, false = NO uint yesPool; // total USDC on YES side uint noPool; // total USDC on NO side uint totalShares; // total shares ever issued } mapping(uint => mapping(address => uint)) public yesShares; mapping(uint => mapping(address => uint)) public noShares; createMarket(question, endTime, seedAmount) → transfers seedAmount USDC from admin to both pools equally → sets initial yesPool = noPool = seedAmount / 2 buyShares(marketId, isYes, usdcAmount) → calls AMMPricer.getShares() to calculate shares → transfers USDC from user to contract → updates yesPool or noPool → mints shares to user mapping → emits SharesBought(marketId, user, isYes, usdcAmount, shares) resolveMarket(marketId, outcome) → owner only, sets resolved + outcome claimWinnings(marketId) → reads user's winning shares → payout = (userShares / totalWinningShares) * totalPool → transfers USDC to user, zeroes out their shares
// AMMPricer.sol — Solidity library // Constant Product AMM pricing logic, separated for clarity and testability // Used by PredictionMarket.sol via: using AMMPricer for AMMPricer.Pool; library AMMPricer { struct Pool { uint yesPool; // USDC on YES side uint noPool; // USDC on NO side } // YES price (0–1 scaled to 1e18) // yesPrice = noPool / (yesPool + noPool) // Logic: more money on YES → YES is "cheap" → NO is scarce → NO costs more function getYesPrice(Pool memory p) → uint (scaled 1e18) // NO price // noPrice = yesPool / (yesPool + noPool) function getNoPrice(Pool memory p) → uint (scaled 1e18) // How many shares does usdcAmount buy? // shares = usdcAmount / price // Example: buying YES at 0.30 with 30 USDC → 100 YES shares function getShares(Pool memory p, uint usdcAmount, bool isYes) → uint shares // Simulate price impact — what will price be AFTER this buy? // Used by frontend to show slippage warning function getPriceAfterBuy(Pool memory p, uint usdcAmount, bool isYes) → uint newPrice // Validate pool invariant (sanity check) // yesPrice + noPrice must always equal 1.0 (1e18) function assertInvariant(Pool memory p) → bool }
// Pricing example (readable) // Initial state (seeded equally) yesPool = 500 USDC, noPool = 500 USDC yesPrice = 500/1000 = 0.50 (50%) noPrice = 500/1000 = 0.50 (50%) // User buys YES with 100 USDC shares = 100 / 0.50 = 200 YES shares yesPool → 600 USDC, noPool stays 500 USDC // New prices after buy yesPrice = 500/1100 = 0.45 (45%) ← YES got cheaper (more supply) noPrice = 600/1100 = 0.55 (55%) ← NO got more expensive (scarcer) // Market resolves YES wins. Total pool = 1100 USDC // User had 200 YES shares out of e.g. 1400 total YES shares payout = (200 / 1400) * 1100 = ~157 USDC (profit: +57 USDC) // Frontend replicates this formula in TypeScript // for instant UI preview BEFORE user signs tx: function getShares(yesPool, noPool, amount, isYes) { const price = isYes ? noPool / (yesPool + noPool) : yesPool / (yesPool + noPool); return amount / price; } function getPriceAfterBuy(yesPool, noPool, amount, isYes) { const newYes = isYes ? yesPool + amount : yesPool; const newNo = isYes ? noPool : noPool + amount; return isYes ? newNo / (newYes + newNo) : newYes / (newYes + newNo); }

// Problem: if two users submit buyShares() in the same block, // the second tx gets fewer shares than the UI showed (price moved). // Solution: add minShares parameter to buyShares() function buyShares( uint marketId, bool isYes, uint usdcAmount, uint minShares // ← slippage guard ) external { uint shares = AMMPricer.getShares(pool, usdcAmount, isYes); require(shares >= minShares, "Slippage: too few shares"); // ... rest of logic } // Frontend: calculate minShares with 1% slippage tolerance const expectedShares = getShares(yesPool, noPool, amount, isYes); const minShares = expectedShares * 0.99; // 1% slippage await contract.buyShares(marketId, isYes, amount, minShares); // UI: show slippage warning if price impact > 5% const priceAfter = getPriceAfterBuy(yesPool, noPool, amount, isYes); const priceImpact = Math.abs(priceAfter - priceBefore) / priceBefore; if (priceImpact > 0.05) showWarning("High price impact: " + (priceImpact*100).toFixed(1) + "%");

// PredictionMarket.sol — all require() guards // createMarket() require(msg.sender == owner, "Not authorized"); require(endTime > block.timestamp, "End time must be in the future"); require(seedAmount > 0, "Seed amount required"); // buyShares() require(market.endTime > block.timestamp, "Market has expired"); require(!market.resolved, "Market already resolved"); require(usdcAmount > 0, "Amount must be greater than zero"); require(shares >= minShares, "Slippage: too few shares received"); // resolveMarket() require(msg.sender == owner, "Not authorized"); require(market.endTime <= block.timestamp, "Market has not expired yet"); require(!market.resolved, "Market already resolved"); // claimWinnings() require(market.resolved, "Market not resolved yet"); require(userWinningShares > 0, "No winning shares to claim"); // Frontend — catch and translate contract errors try { await contract.buyShares(...); } catch (err) { if (err.message.includes("Market has expired")) showToast("This market has closed"); if (err.message.includes("Slippage")) showToast("Price moved too much, try again"); if (err.message.includes("Market already resolved")) showToast("Market is already resolved"); else showToast("Transaction failed: " + err.shortMessage); }

07 Data Storage Strategy blockchain
▾
// Source of truth: blockchain ALL market data lives on-chain permanently — active and resolved. The smart contract is the single source of truth. Markets never get deleted. Resolved markets remain queryable on-chain forever. On-chain (PredictionMarket.sol) - All markets (active + resolved) - All trades / share balances - Pool sizes (yesPool, noPool) - Resolution outcome - User share holdings → claimWinnings() // Off-chain DB (PostgreSQL) — cache layer only - Market metadata: title, description, category, image - Trade history: synced from on-chain events via event listener - Indexed for fast queries / filtering / sorting - DB is NOT the source of truth — it mirrors chain state - If DB is wiped, it can be rebuilt by replaying contract events // Sync strategy - Backend listens to contract events (MarketCreated, SharesBought, MarketResolved) via viem event watcher - On each event → upsert into DB - Market status field: "ACTIVE" | "RESOLVED" | "EXPIRED" - /api/markets?status=resolved → returns resolved markets from DB - /api/markets?status=active → returns active markets from DB - Market detail page always fetches live data from chain for critical values (pools, outcome) + DB for metadata

// Architecture Socket.io server runs alongside Next.js (custom server.ts). Contract event listener → emits Socket.io events → all connected clients receive updates and re-render without page refresh. Events emitted by server market:trade → { marketId, outcome, amount, shares, yesPrice, noPrice, trader } market:resolved → { marketId, outcome } market:created → { marketId, question, category, endDate } market:expired → { marketId } admin:resolve:pending → { marketId, question, endDate } // admin room only // Client usage (React hook) useMarketSocket(marketId) { on('market:trade') → update price chart + activity feed + pools on('market:resolved') → show resolution banner + enable claim button on('market:expired') → lock trading panel, show "Awaiting resolution" badge } useMarketListSocket() { on('market:trade') → update odds % on listing cards on('market:created') → prepend new market card to grid on('market:expired') → update card status badge to "Expired" } useAdminSocket() { // admin only on('admin:resolve:pending') → increment pending resolution counter in admin nav } // Flow — trade 1. User submits trade tx → tx confirmed on-chain 2. Event listener detects SharesBought event 3. Updates DB (trade record, pool sizes) 4. Emits market:trade via Socket.io to room market:{id} 5. All clients in that room update UI instantly // Flow — market expiry 1. Cron job runs every minute, checks markets WHERE endDate <= now AND status = "ACTIVE" 2. For each expired market: update DB status → "EXPIRED" 3. Emit market:expired to room market:{id} → clients lock trading panel 4. Emit market:expired to room market:list → listing cards update badge 5. Emit admin:resolve:pending to room admin → admin sees notification // Rooms - Each market detail page joins room: market:{marketId} - Listing page joins room: market:list - Admin panel joins room: admin (verified by wallet address) - Socket auth: verify wallet signature for Trader/Admin rooms
13 Cron Job — Market Expiry node-cron
▾
// Implemented in server.ts alongside Socket.io and Next.js // Uses node-cron package: npm install node-cron import cron from 'node-cron'; import { prisma } from './lib/prisma'; import { io } from './lib/socket'; // Run every minute: '* * * * *' cron.schedule('* * * * *', async () => { const expired = await prisma.market.findMany({ where: { status: 'ACTIVE', endDate: { lte: new Date() } } }); for (const market of expired) { // 1. Update DB await prisma.market.update({ where: { id: market.id }, data: { status: 'EXPIRED' } }); // 2. Notify all users on that market page io.to(`market:${market.id}`).emit('market:expired', { marketId: market.id }); // 3. Notify listing page io.to('market:list').emit('market:expired', { marketId: market.id }); // 4. Notify admin io.to('admin').emit('admin:resolve:pending', { marketId: market.id, question: market.question, endDate: market.endDate }); } }); // server.ts boot order 1. Create HTTP server 2. Attach Socket.io to HTTP server 3. Start cron job 4. Start Next.js request handler 5. Listen on port 3000
